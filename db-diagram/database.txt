Table friendship {
  user1_id integer [ref: > users.id]
  user2_id integer [ref: > users.id]
  confirmed bool 
  indexes {
    (user1_id, user2_id) [pk] // composite primary key
  }

}

Table users {
  id integer [primary key]
  email varchar
  login varchar
  user_name varchar
  birthday date
  created_at date
}

Table genre {
  id integer [primary key]
  name_genre varchar
}

Table film {
  id integer [primary key]
  name_film varchar
  description varchar
  release_date date
  duration integer
  genre_id integer [ref: > genre.id]
  rating varchar [ ref: > rating.name_rating]
}

Table film_like {
  film_id integer [ref: > film.id]
  user_id integer [Ref:  > users.id]
  indexes {
    (film_id, user_id) [pk] // composite primary key
  }

}

Table rating {
  name_rating varchar [ primary key]
  description varchar
}
